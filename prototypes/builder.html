from freqtrade.strategy import IStrategy
from pandas import DataFrame
import talib.abstract as ta


class MyNewStrategy(IStrategy):
    """
    EMA crossover strategy with RSI filter
    for BTC perpetual futures.
    """

    # ── ROI table ──
    minimal_roi = {
        "0": 0.10,
        "30": 0.05,
        "60": 0.02,
        "120": 0
    }

    # ── Stoploss ──
    stoploss = -0.035

    # ── Trailing stop ──
    trailing_stop = True
    trailing_stop_positive = 0.01
    trailing_stop_positive_offset = 0.02
    trailing_only_offset_is_reached = True

    # ── Config ──
    timeframe = "1h"
    can_short = True

    # ── Protections ──
    @property
    def protections(self):
        return [
            {"method": "StoplossGuard",
             "lookback_period_candles": 1,
             "trade_limit": 3,
             "stop_duration_candles": 30},
            {"method": "MaxDrawdown",
             "lookback_period_candles": 48,
             "max_allowed_drawdown": 0.05,
             "stop_duration_candles": 12},
            {"method": "CooldownPeriod",
             "stop_duration_candles": 5}
        ]

    def populate_indicators(self, df: DataFrame,
                             metadata: dict) -> DataFrame:
        # Trend
        df["ema_20"] = ta.EMA(df, timeperiod=20)
        df["ema_50"] = ta.EMA(df, timeperiod=50)
        df["adx"] = ta.ADX(df, timeperiod=14)

        # Momentum
        df["rsi"] = ta.RSI(df, timeperiod=14)

        # Volatility
        df["atr"] = ta.ATR(df, timeperiod=14)

        return df

    def populate_entry_trend(self, df: DataFrame,
                              metadata: dict) -> DataFrame:
        # Long entry
        df.loc[
            (df["ema_20"] > df["ema_50"]) &
            (df["rsi"] < 70) &
            (df["adx"] > 25),
            "enter_long"
        ] = 1

        # Short entry
        df.loc[
            (df["ema_20"] < df["ema_50"]) &
            (df["rsi"] > 30),
            "enter_short"
        ] = 1

        return df

    def populate_exit_trend(self, df: DataFrame,
                             metadata: dict) -> DataFrame:
        # Exit long
        df.loc[
            (df["ema_20"] < df["ema_50"]),
            "exit_long"
        ] = 1

        return df

    def leverage(self, pair, current_time,
                current_rate, proposed_leverage,
                max_leverage, entry_tag,
                side, **kwargs) -> float:
        return 10.0
